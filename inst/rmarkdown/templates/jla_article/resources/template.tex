$endif$

\Authors{$for(author)$$author.name$\textsuperscript{$author.num$}$sep$, $endfor$} % Authors

\Submitted{$submitted$}

\Published{$published$}

\Volume{$volume$}
\Number{$number$}
\Pages{$pages$}
\Doi{$doi$}
%----------------------------------------------------------------------------------------

%\Notesname{Notes for Research} %If this is a Practitioner Paper

$if(note)$
$for(note)$
\note{$note$}
